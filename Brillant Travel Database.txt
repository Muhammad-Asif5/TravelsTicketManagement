CREATE PROCEDURE sp_get_Balance
	-- Add the parameters for the stored procedure here
	@BankCashReceiveable int
AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT ON;

    -- Insert statements for procedure here
	SELECT       
	 ROUND(SUM(COALESCE ((CASE WHEN t1.DrCr = 'Dr' THEN t1.Amount END), 0) - 
	 COALESCE ((CASE WHEN t1.DrCr = 'Cr' THEN t1.Amount END), 0)),2) AS Balance
FROM            Voucher AS t1
where t1.Bank_ID=@BankCashReceiveable
END

CREATE PROCEDURE sp_View_Amount_Payable

AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT ON;

    -- Insert statements for procedure here
	SELECT        
(CASE WHEN t2.DrCr = 'Cr' THEN t2.VID END) AS Payment_Ref,
 t2.Received_Paid_Date AS Date, 
 t2.Received_Paid AS [Received From],
 t2.Narration, 
 (CASE WHEN t2.DrCr = 'Cr' THEN t2.Amount END) AS Cedit, 
SUM(COALESCE ((CASE WHEN t2.DrCr = 'Dr' THEN t2.Amount END), 0) 
- COALESCE ((CASE WHEN t2.DrCr = 'Cr' THEN t2.Amount END), 0)) AS Balance,
 (CASE WHEN DailyExpenses.Account IS NULL THEN '' ELSE DailyExpenses.Account END) AS Account, 
 (CASE WHEN AirLines.Payable_Code IS NULL THEN '' ELSE AirLines.Payable_Code END) AS Payable_Code,
  t2.User_Name
FROM            Voucher AS t2 LEFT OUTER JOIN
                         AirLines ON t2.Payable_Code = AirLines.Air_ID LEFT OUTER JOIN
                         DailyExpenses ON t2.Exp_Code = DailyExpenses.Exp_Code
WHERE        (t2.DrCr = 'Cr')
GROUP BY t2.DrCr, t2.VID,t2.Received_Paid_Date, t2.Received_Paid, t2.Narration, t2.Amount,
DailyExpenses.Account, AirLines.Payable_Code, t2.User_Name
ORDER BY Date
END

CREATE proc spGet_AP_Summary
AS
BEGIN
SELECT        
AirLines.Payable_Supplier,
SUM(Invoice_Details.Paid2) AS Debit, 
SUM(Invoice_Details.Net_Payable) AS Credit, 
SUM(Invoice_Details.Paid2) - SUM(Invoice_Details.Net_Payable) AS Balance
FROM            Invoice_Details LEFT JOIN
                         AirLines ON Invoice_Details.Payable_Code = AirLines.Air_ID
WHERE (Invoice_Details.Payable_Code IS NOT NULL)
GROUP BY AirLines.Payable_Supplier,Invoice_Details.Payable_Code
ORDER BY Invoice_Details.Payable_Code
END

CREATE proc spGet_BankOrCash
  @BankID int,	@FromDate Datetime = NULL, @ToDate Datetime = NULL
AS
BEGIN
IF @FromDate IS NULL AND @ToDate IS NULL
BEGIN
SELECT        t2.Received_Paid_Date AS Date, 
(CASE WHEN t2.DrCr = 'Dr' THEN t2.VID ELSE '' END) AS Receipt_Ref, (CASE WHEN t2.DrCr = 'Cr' THEN t2.VID ELSE '' END) AS Payment_Ref,
 t2.Narration, t2.Received_Paid, 
 (CASE WHEN t2.DrCr = 'Cr' THEN t2.Amount END) AS Credit, (CASE WHEN t2.DrCr = 'Dr' THEN t2.Amount END) AS Debit,
  SUM(COALESCE ((CASE WHEN t1.DrCr = 'Dr' THEN t1.Amount END), 0) 
- COALESCE ((CASE WHEN t1.DrCr = 'Cr' THEN t1.Amount END), 0)) AS Balance,
 (CASE WHEN Customers.Cust_Code IS NULL THEN '' ELSE Customers.Cust_Code END) AS Cust_Code, 
(CASE WHEN AirLines.Payable_Code IS NULL THEN '' ELSE AirLines.Payable_Code END) AS Payable_Code,
 (CASE WHEN DailyExpenses.Account IS NULL THEN '' ELSE DailyExpenses.Account END) AS Account
FROM            Voucher AS t1 FULL OUTER JOIN
                         Voucher AS t2 ON t1.Bank_ID = t2.Bank_ID AND t1.VID <= t2.VID FULL OUTER JOIN
                         DailyExpenses ON t2.Exp_Code = DailyExpenses.Exp_Code FULL OUTER JOIN
                         Customers ON t2.Cust_Code = Customers.Cust_ID FULL OUTER JOIN
                         AirLines ON t2.Payable_Code = AirLines.Air_ID
WHERE        (t2.Bank_ID = @BankID)
GROUP BY t2.Received_Paid_Date , t2.DrCr, t2.VID, t2.Narration, t2.Received_Paid, t2.Amount,
   Customers.Cust_Code, t2.Payable_Code, AirLines.Payable_Code, DailyExpenses.Account
ORDER BY t2.VID

END

SELECT        t2.Received_Paid_Date AS Date, 
(CASE WHEN t2.DrCr = 'Dr' THEN t2.VID ELSE '' END) AS Receipt_Ref, (CASE WHEN t2.DrCr = 'Cr' THEN t2.VID ELSE '' END) AS Payment_Ref,
 t2.Narration, t2.Received_Paid, 
 (CASE WHEN t2.DrCr = 'Cr' THEN t2.Amount END) AS Credit, (CASE WHEN t2.DrCr = 'Dr' THEN t2.Amount END) AS Debit,
  SUM(COALESCE ((CASE WHEN t1.DrCr = 'Dr' THEN t1.Amount END), 0) 
- COALESCE ((CASE WHEN t1.DrCr = 'Cr' THEN t1.Amount END), 0)) AS Balance,
 (CASE WHEN Customers.Cust_Code IS NULL THEN '' ELSE Customers.Cust_Code END) AS Cust_Code, 
(CASE WHEN AirLines.Payable_Code IS NULL THEN '' ELSE AirLines.Payable_Code END) AS Payable_Code,
 (CASE WHEN DailyExpenses.Account IS NULL THEN '' ELSE DailyExpenses.Account END) AS Account
FROM            Voucher AS t1 FULL OUTER JOIN
                         Voucher AS t2 ON t1.Bank_ID = t2.Bank_ID AND t1.VID <= t2.VID FULL OUTER JOIN
                         DailyExpenses ON t2.Exp_Code = DailyExpenses.Exp_Code FULL OUTER JOIN
                         Customers ON t2.Cust_Code = Customers.Cust_ID FULL OUTER JOIN
                         AirLines ON t2.Payable_Code = AirLines.Air_ID
WHERE        (t2.Bank_ID = @BankID) AND (t2.Received_Paid_Date between @FromDate and @ToDate)
GROUP BY t2.Received_Paid_Date , t2.DrCr, t2.VID, t2.Narration, t2.Received_Paid, t2.Amount,
   Customers.Cust_Code, t2.Payable_Code, AirLines.Payable_Code, DailyExpenses.Account
ORDER BY t2.VID
END
